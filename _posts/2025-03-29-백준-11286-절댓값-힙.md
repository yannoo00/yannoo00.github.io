### 내 코드

{% highlight C++ %} {% raw %}
#include <iostream	
#include <queue	
#include <cmath	

using namespace std;

struct cmp
{
	bool operator()(int a, int b)
	{
		if(abs(a) == abs(b))
			return a 	 b;
		else
			return abs(a) 	 abs(b);
	}
};

priority_queue <int, vector<int	, cmp	 pq;

int main(void)
{

	ios::sync_with_stdio(false); cin.tie(NULL);
	
	int n = 0;
	int printCount = 0;
	int k;
	
	cin 		 n;
	
	for(int i=0; i<n; i++)
	{
		cin 		 k;
		
		if(!k)
		{
			if(pq.empty())
			{
				cout << 0 << endl;
				continue;
			}
			else
			{
				k = pq.top();
				cout << k << endl;
				pq.pop();
			}
		}
			
		else
			pq.push(k);
	}
}

priority queue를 만들고 비교 구조체를 문제가 원하는 대로 절댓값을 비교하는 식으로 만들면 된다.   
나는 sort에 비교 함수를 넣는 것처럼 함수를 넣으려고 시도했는데, 그렇게는 안되고   
**함수 객체** 를 넣어야 사용할 수 있다.  
따라서 구조체를 선언하고 연산자 오버로딩을 해줬다.
