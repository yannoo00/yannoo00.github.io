### 정답 코드

{% highlight C++ %} {% raw %}
#include <iostream	
#include <vector	
#include <algorithm	
#include <set	

using namespace std;

int arr[1000000];
int n;
set<int	 s;

int main(void)
{
	ios::sync_with_stdio(false); cin.tie(NULL);
	
	cin 		 n;
	for(int i=0; i<n; i++)
	{
		cin 		 arr[i];
		
		auto iter = s.lower_bound(arr[i]);
		if(iter!=s.end())
		{
			s.erase(iter);
		}
		s.insert(arr[i]);
	}
	
	cout << n-s.size();
}
