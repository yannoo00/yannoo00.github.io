```C++
#include <iostream>
#include <vector>
#include <algorithm>
#include <deque>

using namespace std;

int n, tmp;
deque<int> d;

int main(void)
{
	ios::sync_with_stdio(false); cin.tie(NULL);
	
	cin >> n;
	
	
	for(int i=0; i<n; i++)
	{
		cin >> tmp;
		if(tmp == 1)
		{
			cin >> tmp;
			d.push_front(tmp);
		}
		else if(tmp == 2)
		{
			cin >> tmp;
			d.push_back(tmp);
		}
		else if(tmp == 3)
		{
			if(d.empty())
				cout << -1 << "\n";
			else
			{
				cout << d.front()<<"\n";
				d.pop_front();
			}
		}
		else if(tmp == 4)
		{
			if(d.empty())
				cout << -1 << "\n";
			else
			{
				cout << d.back() << "\n";
				d.pop_back();
			}			
		}
		else if(tmp == 5)
		{
			cout << d.size() << "\n";
		}
		else if(tmp == 6)
		{
			cout << d.empty() << "\n";
		}
		else if(tmp == 7)
		{
			if(d.empty())
				cout << -1 <<"\n";
			else
			{
				cout << d.front()<<"\n";
			}
		}
		else if(tmp == 8)
		{
			if(d.empty())
				cout << -1 << "\n";
			else
				cout << d.back() << "\n";
		}
	}
}
```
