TypeScript의 Decorator는 디자인 패턴의 그것과 '장식'이라는 의미는 같지만 다른 방식으로 장식해주는 @Decorator를 제공한다.  

```Typescript

// 1. 데코레이터 함수 정의
// 이 함수는 컴파일 타임/로드 타임에 메서드의 정의를 가로챕니다.
function LogExecution(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    // 원래 정의되어 있던 메서드를 백업해 둡니다.
    const originalMethod = descriptor.value;

    // 원래 메서드 대신 실행될 새로운 함수를 덮어씌웁니다.
    descriptor.value = function (...args: any[]) {
        console.log(`[LOG] ${propertyKey} 메서드 호출됨. 인수: ${JSON.stringify(args)}`);
        
        // 부가 기능(로그)을 수행한 뒤, 원래 메서드를 실행합니다.
        const result = originalMethod.apply(this, args);
        
        return result;
    };
}

// 2. 클래스 적용
class CoffeeMachine {
    
    // @ 기호를 사용해 메서드 위에 데코레이터를 붙여주기만 하면 됩니다.
    @LogExecution
    makeCoffee(type: string, size: string) {
        console.log(`${size} 사이즈 ${type} 커피를 추출합니다.`);
    }
}

// 3. 실행 코드
const machine = new CoffeeMachine();

// 호출자는 데코레이터가 있는지 모른 채 평소처럼 호출합니다.
machine.makeCoffee("아메리카노", "Tall");

/* 출력 결과:
[LOG] makeCoffee 메서드 호출됨. 인수: ["아메리카노","Tall"]
Tall 사이즈 아메리카노 커피를 추출합니다.
*/
```
